├─ customer_id (INT, Primary Key)
├─ name (VARCHAR)
├─ email (VARCHAR)
├─ phone (VARCHAR)
├─ country (VARCHAR)
├─ registration_date (DATE)
└─ status (VARCHAR)

PRODUCTS
├─ product_id (INT, Primary Key)
├─ name (VARCHAR)
├─ category (VARCHAR)
├─ price (DECIMAL)
├─ stock_quantity (INT)
├─ created_date (DATE)
└─ status (VARCHAR)

SALES
├─ sale_id (INT, Primary Key)
├─ customer_id (INT, Foreign Key)
├─ product_id (INT, Foreign Key)
├─ quantity (INT)
├─ unit_price (DECIMAL)
├─ total_amount (DECIMAL)
├─ sale_date (DATE)
└─ payment_status (VARCHAR)

INVENTORY_LOGS
├─ log_id (INT, Primary Key)
├─ product_id (INT, Foreign Key)
├─ quantity_change (INT)
├─ transaction_type (VARCHAR)
└─ log_date (DATETIME)

REVIEWS
├─ review_id (INT, Primary Key)
├─ customer_id (INT, Foreign Key)
├─ product_id (INT, Foreign Key)
├─ rating (INT, 1-5)
├─ review_text (TEXT)
└─ review_date (DATE)
